/**
JavaScript object data type.
@module root
@class Object
*/
/**
Clones the object and all it contains and returns the clone.
@method clone
@returns {Object} clone of the original object	
*/

Object.prototype.clone = function() {
	var newObj = (this instanceof Array) ? [] : {};
	for (var i in this) {
		if (i == 'clone') continue;
		if (this[i] && typeof this[i] == "object") newObj[i] = this[i].cloneNode ? this[i].cloneNode() : this[i].clone();
		else newObj[i] = this[i]
	}
	if(newObj && newObj.onCloneInit) newObj.onCloneInit();
	return newObj;
};

/**
Extend the object with functions and fields given.
@method extend
@param data {Object} JSON which should be added to the current object.
@returns {Object} original object plus all the content from the given data object.
*/

Object.prototype.extend = function(data) {   
	for (var property in data)
		this[property] = data[property];
};

/**
JavaScript Date data type.
@class Date
*/
/**
Clones the Date and returns the clone.
@method clone
@returns {Date} Clone of the Date.
*/
Date.prototype.clone = function()
{
	var copy = new Date();
	copy.setTime(this.getTime());
	return copy;
};

/**
Formats money to a more human-readable form and returns it.
@method formatMoney
@param c {Number} Number of decimals to round at.
@param d {String} Character set between numbers to keep them readable. (3000000 -> 3[d]000[d]000) Defaults to ","
@param t {String} Character set between full numbers and the decimal fractions. (1.5 -> 1[t]5) Defaults to "."
@returns {String} Formatted money as string.
*/
Number.prototype.formatMoney = function(c, d, t){
  var n = this, c = isNaN(c = Math.abs(c)) ? 2 : c, d = d == undefined ? "," : d, t = t == undefined ? "." : t, s = n < 0 ? "-" : "", i = parseInt(n = Math.abs(+n || 0).toFixed(c)) + "", j = (j = i.length) > 3 ? j % 3 : 0;
  return s + (j ? i.substr(0, j) + t : "") + i.substr(j).replace(/(\d{3})(?=\d)/g, "$1" + t) + (c ? d + Math.abs(n - i).toFixed(c).slice(2) : "");
};

/**
@class Array
*/
/**
Inserts an item into an array at the given index
@method insert
@param index {Number} position in the array the new item should take place.
@param item {Object} the new item to add.
*/
Array.prototype.insert = function (index, item) {
  if(index >= this.length)
  	this[this.length] = item;
  else
    this.splice(index, 0, item);
};
