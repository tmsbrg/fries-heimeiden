Controller element
@module Controller
*/
/**
Controller element of the HTML5_2DREngine.
@class Controller
*/
Controller = {

	/**
	True if control key is pressed down.
	@property isCtrl
	@type Boolean
	*/
	isCtrl 			: false,
	/**
	Array of all the keycodes and their boolean state.
	@property keyDown
	@type Array
	*/
	keyDown 		: [],
	
	/**
	Initializes the Controller.
	@method init
	@returns {Boolean} True when everything has been loaded.
	*/
	init			:	function()
	{
		document.onkeyup=function(e){
			if(e.which == 17)
				Controller.isCtrl=false;
			
			Controller.keyDown[e.which] = false;
		}
		
		document.onkeydown=function(e){
			if(e.which == 17)
				Controller.isCtrl=true;
			
			Controller.keyDown[e.which] = true;
		}
		return true;
	},
	
	/**
	update function will be called before every draw call.
	@method base_update
	*/
	base_update			:	function()
	{
		for(var i = 0; i < Model.drawableObjects.length; i++)
			if(Model.drawableObjects[i])
				Model.drawableObjects[i].base_update();
		
		if(this.update)
			this.update();
	},
	
	/**
	update function will be called before every draw call.
	Most of the game logic will be put in here.
	@method update
	*/
	update				:	function()
	{
		
	},
	
	/**
	will be called by the <View> when an object has been pressed with the mouse button.
	@method mouse_objectClick
	@param object {BaseDrawable} Object that has been pressed by the mouse button.
	*/
	mouse_objectClick		:	function(object)
	{
		object.onclick();
	},
	
	/**
	will be called by the <View> when an object has been double clicked with the mouse button.
	method mouse_objectDoubleClick
	@param object {BaseDrawable} Object that has been double clicked by the mouse button.
	*/
	mouse_objectDoubleClick		:	function(object)
	{
		object.onclick();
	},
	
	/**
	will be called by the <View> when an object is being hovered the mouse.
	@method mouse_objectHover
	@param object {BaseDrawable} Object that is being hovered by the mouse.
	*/
	mouse_objectHover		:	function(object)
	{
		object.onhover();
	},
	
	/**
	will be called by the <View> when an object is not being hovered anymore by the mouse.
	method mouse_objectHoverOut	   
	@param object {BaseDrawable} Object that is not being hovered anymore by the mouse.
	*/
	mouse_objectHoverOut	:	function(object)
	{
		object.onmouseout();
	}	
}
